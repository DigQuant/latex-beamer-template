            \begin{iCodeEnv}{something\_analysis.py}{python}
import numpy as np
import pandas as pd

def calculate_metrics(data):
    """计算核心指标"""
    mean_val = np.mean(data)
    std_val = np.std(data)
    return {
        'mean': mean_val,
        'std': std_val,
        'z_score': (data - mean_val) / std_val
    }

# 加载数据
df = pd.read_csv('sales_data.csv')
metrics = calculate_metrics(df['amount'])
            \end{iCodeEnv}

        scale=0.8, transform shape,
        node distance=0.5cm and 1.5cm,
        box/.style={rectangle, draw=main, fill=main!5, thick, minimum width=2.5cm, minimum height=1cm, rounded corners, align=center},
        decision/.style={diamond, draw=alert, fill=alert!5, thick, aspect=2, minimum width=2.5cm, align=center, font=\footnotesize},
        line/.style={-Stealth, thick, draw=black!70}
    ]
        % Nodes
        \node[box] (start) {开始};
        \node[box, below=of start] (step1) {数据预处理};
        \node[decision, below=of step1] (check) {数据完整?};
        \node[box, right=of check] (fix) {修复数据};
        \node[box, below=of check, yshift=-0.5cm] (analyze) {进行分析};
        \node[box, below=of analyze] (end) {生成报告};

        % Edges
        \draw[line] (start) -- (step1);
        \draw[line] (step1) -- (check);
        \draw[line] (check) -- node[right, font=\tiny] {No} (fix);
        \draw[line] (fix) |- (step1);
        \draw[line] (check) -- node[right, font=\tiny] {Yes} (analyze);
        \draw[line] (analyze) -- (end);
    \end{tikzpicture}
